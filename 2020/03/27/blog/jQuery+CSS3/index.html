# 前言

一款 jQuery+CSS3 的文字背景粒子动画特效，一共 6 种粒子效果，每种文字背景的粒子效果都不同，有漂浮的有坠落的等等。

<script>
function fire() {
.each(.each(.each((".particletext.fire"), function(){
var firecount = ($(this).width()/50)*20;
for(var i = 0; i <= firecount; i++) {
var size = $.rnd(8,12);
$(this).append('<span class="particle" style="top:' + $.rnd(40,70) + '%; left:' + .rnd(−10,100)+′.rnd(-10,100) + '%;width:' + size + 'px; height:' + size + 'px;animation-delay: ' + (.rnd(−10,100)+′.rnd(0,20)/10) + 's;"></span>');
}
});
}

</script>

function initparticles() {

jQuery.rnd = function(m,n) {
m = parseInt(m);
n = parseInt(n);
return Math.floor( Math.random() * (n - m + 1) ) + m;
}
initparticles();
</script>
